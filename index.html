    document.addEventListener('DOMContentLoaded', function () {

        const grid = document.querySelector('.projects_grid');

        const projects = Array.from(document.querySelectorAll('.project_card:not(.show-more-card):not(.hide-card)'));

        const itemsToShow = 5;


        if (projects.length > itemsToShow) {

        function showCollapsedState() {

            removeGeneratedCards();

            const hiddenCount = projects.length - itemsToShow;

            projects.forEach((project, index) => {

                if (index >= itemsToShow) {

                    project.style.display = 'none';

                    project.style.animation = '';

                    project.style.opacity = '';

                } else {

                    project.style.display = 'flex';

                    project.style.animation = `fadeIn 0.5s ease ${index * 0.1}s forwards`;

                    project.style.opacity = '0';

                }

            });

            const card = createActionCard('show', hiddenCount);

            card.addEventListener('click', showExpandedState);

            grid.appendChild(card);

        }


        function showExpandedState() {

            removeGeneratedCards();

            projects.forEach((project, index) => {

                if (project.style.display === 'none') {

                    project.style.display = 'flex';

                    project.style.animation = `fadeIn 0.5s ease ${(index - itemsToShow) * 0.1}s forwards`;

                    project.style.opacity = '0';

                }

            });

            const card = createActionCard('hide');

            card.addEventListener('click', () => {

                showCollapsedState();

                document.getElementById('projects').scrollIntoView({ behavior: 'smooth' });

            });

            grid.appendChild(card);

        }
